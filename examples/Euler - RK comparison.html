def f(dt, Analytical, Euler, Runge_Kutta):
    n = int(100/dt)
    args = {'m':1, 'c':0.1, 'k':1, 'x0':1, 'v0':0, 'n':n, 'dt':dt}
    a = sdof_analytical(**args)

    fig1 = plt.figure()
    ax1 = fig1.add_subplot(111)
